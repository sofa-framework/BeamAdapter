31 enum class BeamAdapterAction
32 {
33 NO_ACTION = 0,
34 MOVE_FORWARD,
35 MOVE_BACKWARD,
36 SPIN_RIGHT,
37 SPIN_LEFT,
38 SWITCH_NEXT_TOOL,
39 SWITCH_PREVIOUS_TOOL,
40 DROP_TOOL,
41 USE_TOOL_0,
42 USE_TOOL_1,
43 USE_TOOL_2,
44 };
45
47 const static std::map<std::string, BeamAdapterAction> beamActionNames = {
48 {"stop", BeamAdapterAction::NO_ACTION},
49 {"fwd", BeamAdapterAction::MOVE_FORWARD},
50 {"bwd", BeamAdapterAction::MOVE_BACKWARD},
51 {"right", BeamAdapterAction::SPIN_RIGHT},
52 {"nextT", BeamAdapterAction::SWITCH_NEXT_TOOL},
53 {"prevT", BeamAdapterAction::SWITCH_PREVIOUS_TOOL},
54 {"dropT", BeamAdapterAction::DROP_TOOL},
55 {"tool0", BeamAdapterAction::USE_TOOL_0},
56 {"tool1", BeamAdapterAction::USE_TOOL_1},
57 {"tool2", BeamAdapterAction::USE_TOOL_2}
58 };
59
61 inline BeamAdapterAction convertBeamAdapterAction(const std::string& sAction)
62 {
63 auto bAction = beamActionNames.find(sAction);
64
65 if (bAction != beamActionNames.end())
66 return bAction->second;
67 else
68 return BeamAdapterAction::NO_ACTION;
69 }
