37template <class DataTypes>
38void BaseBeamInterpolation<DataTypes>::getControlPointsFromFrame(const Transform& global_H_local0, const Transform& global_H_local1,
39 const Real L,
40 Vec3& P0, Vec3& P1,
41 Vec3& P2, Vec3& P3)
42{
43 P0 = global_H_local0.getOrigin();
44 P3 = global_H_local1.getOrigin();
45
46 P1 = P0 + global_H_local0.getOrientation().rotate(Vec3(1.0, 0, 0)) * (L / 3.0);
47 P2 = P3 + global_H_local1.getOrientation().rotate(Vec3(-1, 0, 0)) * (L / 3.0);
48}

59template <class DataTypes>
60void BaseBeamInterpolation<DataTypes>::RotateFrameForAlignNormalizedX(const Quat& input, const Vec3& x, Quat& output)
61{
62 const Vec3 x0 = input.inverseRotate(x);
63
64 const Real cTheta = x0[0];
65 if (cTheta > 0.9999999999)
66 {
67 output = input;
68 }
69 else
70 {
71 // axis of rotation
72 Vec3 dw(0, -x0[2], x0[1]);
73 dw.normalize();
74
75 // optimized axisToQuat with no normalization, using already computed cos(Theta) and x being 0
76 constexpr auto optimAxisToQuat = [](const Vec3& v, Real cosTheta)
77 {
78 Quat q(type::QNOINIT);
79
80 const auto sp = std::sqrt(static_cast<Real>(0.5) * (static_cast<Real>(1) - cosTheta)); //sin(phi/2) = sqrt(0.5*(1+cos(phi)))
81 const auto cp = std::sqrt(static_cast<Real>(0.5) * (static_cast<Real>(1) + cosTheta)); //cos(phi/2) = sqrt(0.5*(1-cos(phi)))
82 q[0] = static_cast<Real>(0.0);
83 q[1] = v.y() * sp;
84 q[2] = v.z() * sp;
85 q[3] = cp;
86
87 return q;
88 };
89
90 // computation of the rotation
91 const Quat inputRoutput = optimAxisToQuat(dw, cTheta);
92
93 output = input * inputRoutput;
94 }
95}

167template<class DataTypes>
168void BaseBeamInterpolation<DataTypes>::addBeam(const EdgeID eID, const Real length, const Real x0, const Real x1, const Real angle)
169{
170 auto edgeList = sofa::helper::getWriteOnlyAccessor(d_edgeList);
171 auto lengthList = sofa::helper::getWriteOnlyAccessor(d_lengthList);
172 auto DOF0TransformNode0 = sofa::helper::getWriteOnlyAccessor(d_DOF0TransformNode0);
173 auto DOF1TransformNode1 = sofa::helper::getWriteOnlyAccessor(d_DOF1TransformNode1);
174 auto curvAbsList = sofa::helper::getWriteOnlyAccessor(d_curvAbsList);
175
176 curvAbsList.push_back(Vec2(x0, x1));
177
178 edgeList.push_back(eID);
179 lengthList.push_back(length);
180
181 Quat QuatX;
182 QuatX.axisToQuat(Vec3(1, 0, 0), angle);
183 QuatX.normalize();
184
185 // as an angle is set between DOFs and Beam, they are no more aligned
186 d_dofsAndBeamsAligned.setValue(false);
187 DOF0TransformNode0.push_back(Transform(Vec3(0, 0, 0), QuatX));
188 DOF1TransformNode1.push_back(Transform(Vec3(0, 0, 0), QuatX));
189}

256template<class DataTypes>
257void BaseBeamInterpolation<DataTypes>::getBeamAtCurvAbs(const Real x_input, unsigned int& edgeInList_output, Real& baryCoord_output, unsigned int start)
258{
260 Real lTotalRest = getRestTotalLength();
262 Real LTotal = 0.0;
263
264 const sofa::Size edgeListSize = static_cast<sofa::Size>(this->d_edgeList.getValue().size());
265
267 for (unsigned int e = start; e < edgeListSize; e++)
268 {
269 LTotal += this->getLength(e);
270 }
271
273 Real x_i = x_input + LTotal - lTotalRest;
274
275 if (x_i < 0.0)
276 {
277 edgeInList_output = start;
278 baryCoord_output = 0;
279 return;
280 }
281
283 Real x = 0;
284
285 for (unsigned int e = start; e < edgeListSize; e++)
286 {
287 x += this->getLength(e);
288 if (x > x_i)
289 {
290 edgeInList_output = e;
291 Real x0 = x - this->getLength(e);
292 baryCoord_output = (x_i - x0) / this->getLength(e);
293 return;
294 }
295 }
296
297 edgeInList_output = edgeListSize - 1;
298 baryCoord_output = 1.0;
299}

302template<class DataTypes>
303int BaseBeamInterpolation<DataTypes>::computeTransform(const EdgeID edgeInList, const PointID node0Idx, const PointID node1Idx, Transform& global_H_local0, Transform& global_H_local1, const VecCoord& x)
304{
306 Transform DOF0_H_local0, DOF1_H_local1;
307 getDOFtoLocalTransform(edgeInList, DOF0_H_local0, DOF1_H_local1);
308
310 Transform global_H_DOF0(x[node0Idx].getCenter(), x[node0Idx].getOrientation());
311 Transform global_H_DOF1(x[node1Idx].getCenter(), x[node1Idx].getOrientation());
313 global_H_local0 = global_H_DOF0 * DOF0_H_local0;
314 global_H_local1 = global_H_DOF1 * DOF1_H_local1;
315
316 return 1;
317}

335template<class DataTypes>
336void BaseBeamInterpolation<DataTypes>::getDOFtoLocalTransformInGlobalFrame(const EdgeID edgeInList, Transform& DOF0Global_H_local0, Transform& DOF1Global_H_local1, const VecCoord& x)
337{
338
339 Transform DOF0_H_local0, DOF1_H_local1;
340 getDOFtoLocalTransform(edgeInList, DOF0_H_local0, DOF1_H_local1);
341
342 unsigned int node0Idx, node1Idx;
343 getNodeIndices(edgeInList, node0Idx, node1Idx);
344
345 // Computes the Rotation to global from DOF0 and DOF1
346 Transform global_R_DOF0(Vec3(0, 0, 0), x[node0Idx].getOrientation());
347 Transform global_R_DOF1(Vec3(0, 0, 0), x[node1Idx].getOrientation());
348
349 // - rotation due to the optional transformation
350 DOF0Global_H_local0 = global_R_DOF0 * DOF0_H_local0;
351 DOF1Global_H_local1 = global_R_DOF1 * DOF1_H_local1;

378template<class DataTypes>
379void BaseBeamInterpolation<DataTypes>::getTangent(Vec3& t, const Real baryCoord,
380 const Transform& global_H_local0,
381 const Transform& global_H_local1, const Real L)
382{
383 Vec3 P0, P1, P2, P3;
384
385 this->getControlPointsFromFrame(global_H_local0, global_H_local1, L, P0, P1, P2, P3);
386
387 t = P0 * (-3 * (1 - baryCoord) * (1 - baryCoord))
388 + P1 * (3 - 12 * baryCoord + 9 * baryCoord * baryCoord)
389 + P2 * (6 * baryCoord - 9 * baryCoord * baryCoord) + P3 * (3 * baryCoord * baryCoord);
390
391 t.normalize();
392}

490template<class DataTypes>
491void BaseBeamInterpolation<DataTypes>::computeStrechAndTwist(const EdgeID edgeInList, const VecCoord& x, Vec3& ResultNodeO, Vec3& ResultNode1)
492{
493
495
497 Vec3 P0, P1, P2, P3;
498 getSplinePoints(edgeInList, x, P0, P1, P2, P3);
499
501 unsigned int node0Idx, node1Idx;
502 dmsg_info() << "in computeStrechAndTwist";
503 getNodeIndices(edgeInList, node0Idx, node1Idx);
504
505 Real length0, length1;
506 length0 = 0.0;
507 length1 = 0.0;
508
509 Vec3 seg;
510 Vec3 pos = P0;
511 Vec3 n_x, n_y, n_z, x_b, y_b, z_b;
512 Quat R0 = x[node0Idx].getOrientation();
513 R0.normalize();
514 n_x = R0.rotate(Vec3(1.0, 0.0, 0.0));
515 n_y = R0.rotate(Vec3(0.0, 1.0, 0.0));
516 n_z = R0.rotate(Vec3(0.0, 0.0, 1.0));
517
518 for (Real bx = 0.02; bx < 1.00001; bx += 0.02)
519 {
521 seg = -pos;
522 pos = P0 * (1 - bx) * (1 - bx) * (1 - bx) + P1 * 3 * bx * (1 - bx) * (1 - bx) + P2 * 3 * bx * bx * (1 - bx) + P3 * bx * bx * bx;
523 seg += pos;
524 if (bx < 0.50001)
525 length0 += seg.norm();
526 else
527 length1 += seg.norm();
528
530 n_x = P0 * (-3 * (1 - bx) * (1 - bx)) + P1 * (3 - 12 * bx + 9 * bx * bx) + P2 * (6 * bx - 9 * bx * bx) + P3 * (3 * bx * bx);
531 n_x.normalize();
532 n_z = n_x.cross(n_y);
533 n_z.normalize();
534 n_y = n_z.cross(n_x);
535 n_y.normalize();
536
537 if (bx > 0.49999 && bx < 0.50001)
538 {
540 x_b = n_x;
541 y_b = n_y;
542 z_b = n_z;
543 }
544 }
545
547 Quat globalRgeom1;
548 globalRgeom1 = globalRgeom1.createQuaterFromFrame(n_x, n_y, n_z);
549 Vec3 y_geom1 = globalRgeom1.rotate(Vec3(0.0, 1.0, 0.0));
550 Vec3 z_geom1 = globalRgeom1.rotate(Vec3(0.0, 0.0, 1.0));
551
552 Vec3 x_1, y_1;
553 Quat R1 = x[node1Idx].getOrientation();
554 R1.normalize();
555 x_1 = R1.rotate(Vec3(1.0, 0.0, 0.0));
556 y_1 = R1.rotate(Vec3(0.0, 1.0, 0.0));
557
558
560 Real cosTheta = y_1 * y_geom1;
561 Real theta;
562 if (cosTheta > 1.0)
563 theta = 0.0;
564 else
565 {
566 if (y_1 * z_geom1 < 0.0)
567 theta = -acos(cosTheta);
568 else
569 theta = acos(cosTheta);
570 }
571
572 ResultNodeO[0] = length0; ResultNodeO[1] = 0.0; ResultNodeO[2] = theta / 2.0;
573 ResultNode1[0] = length1; ResultNode1[1] = 0.0; ResultNode1[2] = theta / 2.0;
574}

594template<class DataTypes>
595void BaseBeamInterpolation<DataTypes>::updateBezierPoints(const VecCoord& x, unsigned int index,
596 VectorVec3& bezierPosVec)
597{
598 const Real _L = d_lengthList.getValue()[index];
599
602 Transform global_H_local0, global_H_local1;
603 computeTransform(index, global_H_local0, global_H_local1, x);
604

607 bezierPosVec[index * 4 + 3] = global_H_local1.getOrigin(); //P3
608 bezierPosVec[index * 4 + 1] = bezierPosVec[index * 4] + global_H_local0.getOrientation().rotate(Vec3(1.0, 0, 0)) * (_L / 3.0); //P1
609 bezierPosVec[index * 4 + 2] = bezierPosVec[index * 4 + 3] + global_H_local1.getOrientation().rotate(Vec3(-1, 0, 0)) * (_L / 3.0); //P2

657 }
658 }
659 else
660 {
662 const Real _L = d_lengthList.getValue()[edgeInList];

665 {
666 Vec3 x_local(0, 0, 0);

668
669 InterpolateTransformUsingSpline(edgeInList, baryCoord, x_local, x, global_H_localInterpol);
670 posResult = global_H_localInterpol.getOrigin() + global_H_localInterpol.getOrientation().rotate(localPos);
671 }
672 else
673 {
674 const VectorVec3& v = m_StateNodes->read(vecXId)->getValue();
675 Vec3 P0, P1, P2, P3;
676
677 P0 = v[edgeInList * 4];
678 P1 = v[edgeInList * 4 + 1];
679 P2 = v[edgeInList * 4 + 2];
680 P3 = v[edgeInList * 4 + 3];
681
682 Vec3 DP = P0 - P3;
683
684 if (DP.norm() < _L * 0.01)
685 {
686 posResult = P0;
687 return;
688 }
689
690 Real bx = baryCoord;
691
692 posResult = P0 * (1 - bx) * (1 - bx) * (1 - bx) + P1 * 3 * bx * (1 - bx) * (1 - bx) + P2 * 3 * bx * bx * (1 - bx) + P3 * bx * bx * bx;
693 }

725template<class DataTypes>
726void BaseBeamInterpolation<DataTypes>::InterpolateTransformUsingSpline(Transform& global_H_localResult,
727 const Real baryCoord,
728 const Transform& global_H_local0,
729 const Transform& global_H_local1,
730 const Real L)
731{
732 Vec3NoInit P0, P1, P2, P3;
733
735 this->getControlPointsFromFrame(global_H_local0, global_H_local1, L, P0, P1, P2, P3);
736
737 const Real invBx = 1 - baryCoord;
738 Vec3NoInit posResult;
739 Quat quatResult;
740
741 const Vec3 dP01 = P1 - P0;
742 const Vec3 dP12 = P2 - P1;
743 const Vec3 dP03 = P3 - P0;
744
745 if (dP01 * dP12 < 0.0 && dP03.norm() < 0.4 * L)
746 {
751 quatResult = global_H_local0.getOrientation();
752
753 //quatResult.slerp(global_H_local0.getOrientation(),global_H_local1.getOrientation(),bx,true);
754 posResult = P0 * invBx + P3 * baryCoord;
755 }
756 else
757 {
758 const Real bx2 = baryCoord * baryCoord;
759 const Real invBx2 = invBx * invBx;
760
762 posResult = P0 * invBx * invBx2 + P1 * 3 * baryCoord * invBx2 + P2 * 3 * bx2 * invBx + P3 * bx2 * baryCoord;
763
765 Vec3 n_x = P0 * (-3 * invBx2) + P1 * (3 - 12 * baryCoord + 9 * bx2) + P2 * (6 * baryCoord - 9 * bx2) + P3 * (3 * bx2);
766 n_x.normalize();
767
770 Quat R0, R1, Rslerp;
771
773 RotateFrameForAlignNormalizedX(global_H_local0.getOrientation(), n_x, R0);
774 RotateFrameForAlignNormalizedX(global_H_local1.getOrientation(), n_x, R1);
775
777 Rslerp.slerp(R0, R1, (float)baryCoord, true);
778 Rslerp.normalize();
779
781 RotateFrameForAlignNormalizedX(Rslerp, n_x, quatResult);
782 }
783
784 global_H_localResult.set(posResult, quatResult);
785}

854template<class DataTypes>
855typename BaseBeamInterpolation<DataTypes>::Real BaseBeamInterpolation<DataTypes>::ComputeTotalBendingRotationAngle(const Real dx_computation,
856 const Transform& global_H_local0, const Transform& global_H_local1, const Real L,
857 const Real baryCoordMin, const Real baryCoordMax)
858{
859 Vec3 P0, P1, P2, P3, t0, t1;
860
861 this->getControlPointsFromFrame(global_H_local0, global_H_local1, L, P0, P1, P2, P3);
862
863 t0 = P0 * (-3 * (1 - baryCoordMin) * (1 - baryCoordMin)) +
864 P1 * (3 - 12 * baryCoordMin + 9 * baryCoordMin * baryCoordMin) +
865 P2 * (6 * baryCoordMin - 9 * baryCoordMin * baryCoordMin) +
866 P3 * (3 * baryCoordMin * baryCoordMin);
867 t0.normalize();
868
869 Real BendingAngle = 0.0;
870
871 unsigned int numComputationPoints = (unsigned int)ceil((baryCoordMax - baryCoordMin) * (L / dx_computation));
872 numComputationPoints = std::max(numComputationPoints, 3u);
873
874 for (unsigned int i = 0; i < numComputationPoints; i++)
875 {
876 Real bx = ((Real)((i + 1) / numComputationPoints)) * (baryCoordMax - baryCoordMin) + baryCoordMin;
877 t1 = P0 * (-3 * (1 - bx) * (1 - bx)) + P1 * (3 - 12 * bx + 9 * bx * bx) + P2 * (6 * bx - 9 * bx * bx) + P3 * (3 * bx * bx);
878 t1.normalize();
879
880 if (dot(t0, t1) < 1.0)
881 BendingAngle += acos(dot(t0, t1));
882
883 t0 = t1;
884 }
885
886 return BendingAngle;
887}
