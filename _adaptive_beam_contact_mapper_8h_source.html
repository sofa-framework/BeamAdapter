102 int addBaryPoint(const Vector3& baryP, int splineId, Real& r)
103 {
104 int i = nbp++;
105 if ((int)outmodel->getX()->size() <= i)
106 outmodel->resize(i+1);
107 if (mapping)
108 {
109 CubicBezierCurve<1> cb(this->model, splineId);
110 int localsplineId;
111 int edgeId = model->comuteEdgeIdFromSplineId(splineId,localsplineId);
112 if ( cb.isCubicBezier() )
113 i = mapping->addBaryPoint(edgeId,baryP,true);
114 else
115 i = mapping->addBaryPoint(edgeId,baryP,false);
116 r=cb.r();
117 }
118 else
119 {
120 std::cout << "WARNING[AdaptiveBeamContactMapper] AdaptiveBeamMapping is not created" << std::endl;
121 }
122 return i;
123 }

145 }
