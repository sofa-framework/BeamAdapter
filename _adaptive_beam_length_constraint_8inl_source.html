93template<class DataTypes>
94void AdaptiveBeamLengthConstraint<DataTypes>::detectElongation(const VecCoord& x, const VecCoord& xfree)
95{
96 Vec3 P0,P1,P2,P3;
97 Real length, rest_length, rest_length_interval;
98 Real alarmLength = m_alarmLength.getValue();
99 bool prev_stretch = false;
100
101 auto* interpolation = m_interpolation.get();
102
105 rest_length_interval=0.0;
106
108 IntervalDefinition<Real> intervalDef;
109 Real angleInterval=0.0;
110
111 for (unsigned int b=0; b<interpolation->getNumBeams(); b++)
112 {
114 interpolation->getSplinePoints(b,x,P0,P1,P2,P3);
115
116 //TODO(dmarchal 2017) Please specify who/when this will be done
118 length=(P0-P3).norm();
119 rest_length = interpolation->getLength(b);
120
121 unsigned n0, n1;
122 interpolation->getNodeIndices(b, n0, n1);
123
125 Transform Tnode0, Tnode1;
126 interpolation->computeTransform(b, n0, n1, Tnode0,Tnode1,x);
127 Real angleBeam = interpolation->ComputeTotalBendingRotationAngle(rest_length/10.0, Tnode0, Tnode1,rest_length , 0.0, 1.0);
128
130 bool case1a = (n0==n1);
131
132 if(prev_stretch)
133 {
141
142 bool case1b = (rest_length*alarmLength > length);
143 bool case1c = (angleBeam >= 0.99*m_maxBendingAngle.getValue());
144 case1c=false;
145 case1b=false;
146
148 if (case1a || case1b || case1c)
149 {
150
151 dmsg_info() <<" beam "<<b<<" case 1 detected (a):"<<n0<<" == ?"<<n1<<" (b) :"<<
152 rest_length*alarmLength<<" > ?"<<length<<" (c) : "<<angleBeam+angleInterval<< " > ?" <<m_maxBendingAngle.getValue() ;
153
156 intervalDef.posEnd = P0;
157 intervalDef.IdxEnd = n0;
158
159 Transform DOF0_H_local0, DOF1_H_local1;
160 interpolation->getDOFtoLocalTransform(b, DOF0_H_local0, DOF1_H_local1);
161
162 intervalDef.dof_H_end = DOF0_H_local0;
163
164 interpolation->getSplinePoints(b,xfree,P0,P1,P2,P3);
165
166 Transform global_H_local0_free, global_H_local1_free;
167 interpolation->computeTransform(b, n0, n1, global_H_local0_free, global_H_local1_free, xfree);
168 intervalDef.posFreeEnd = global_H_local0_free.getOrigin();
169
170 intervalDef.rest_length=rest_length_interval;
171
172 dmsg_info() << " rest_length_interval ="<<rest_length_interval ;
173
175 prev_stretch=false;
176 angleInterval=0.0;
177 rest_length_interval=0.0;
178
180 if ((intervalDef.posBegin-intervalDef.posEnd).norm() < 0.0000001)
181 msg_warning() <<"interval of size = 0 detected => 1 beam has a bendingAngle > m_maxBendingAngle" ;
182 else
183 m_constraintIntervals.push_back(intervalDef);
184
186 if (!case1a && !case1b && case1c) //
187 {
188 dmsg_info() <<" isolate case 1(c)" ;
189
191 prev_stretch=true;
192 angleInterval = angleBeam;
193 rest_length_interval = rest_length;
194
197 intervalDef.dof_H_begin = DOF0_H_local0;
198 intervalDef.IdxBegin = n0;
199 intervalDef.posBegin = intervalDef.posEnd;
200 intervalDef.posFreeBegin = intervalDef.posFreeEnd;
201 }
202 }
203 else
204 {
206 angleInterval+=angleBeam;
207 rest_length_interval += rest_length;
208 }
209 }
210 else
211 {
216 bool case2a = (rest_length*alarmLength < length);
217 case2a=true;
218
219 if ( case2a && !case1a )
220 {
221 dmsg_info() << " beam "<<b<<" case 2 (a) detected "<<rest_length*alarmLength<<" < ?"<<length ;
222
224 prev_stretch=true;
225 angleInterval = angleBeam;
226 rest_length_interval = rest_length;
227
230 Transform DOF0_H_local0, DOF1_H_local1;
231 interpolation->getDOFtoLocalTransform(b, DOF0_H_local0, DOF1_H_local1);
232
233 Transform global_H_local0, global_H_local1;
234 interpolation->computeTransform(b, n0, n1, global_H_local0, global_H_local1, x);
235
236 Transform global_H_local0_free, global_H_local1_free;
237 interpolation->computeTransform(b, n0, n1, global_H_local0_free, global_H_local1_free, xfree);
238
239 intervalDef.dof_H_begin = DOF0_H_local0;
240 intervalDef.IdxBegin = n0;
241 intervalDef.posBegin = global_H_local0.getOrigin();
242 intervalDef.posFreeBegin = global_H_local0_free.getOrigin();
243 }
244 else
245 {
246 dmsg_info() <<" beam "<<b<<" case 2 (b) detected "<<rest_length*alarmLength<<" > ?"<<length<<" or n0="<<n0<<" ==? "<<"n1="<<n1 ;
247 }
248 }
249 }
250
251 unsigned int b = interpolation->getNumBeams()-1;
252 unsigned n0, n1;
253 interpolation->getNodeIndices(b, n0, n1);
254 if(prev_stretch)
255 {
256 dmsg_info() <<" case 1 (d) detected on the last beam" ;
257
260 Transform DOF0_H_local0, DOF1_H_local1;
261 interpolation->getDOFtoLocalTransform(b, DOF0_H_local0, DOF1_H_local1);
262
263 Transform global_H_local0, global_H_local1;
264
265 interpolation->computeTransform(b, n0, n1, global_H_local0, global_H_local1, x);
266
267
268 Transform global_H_local0_free, global_H_local1_free;
269 interpolation->computeTransform(b, n0, n1, global_H_local0_free, global_H_local1_free, xfree);
270
271
272 intervalDef.dof_H_end = DOF1_H_local1;
273 intervalDef.IdxEnd = n1;
274 intervalDef.posEnd = global_H_local1.getOrigin();
275 intervalDef.posFreeEnd = global_H_local1_free.getOrigin();
276 intervalDef.rest_length=rest_length_interval;
277
278 dmsg_info() <<" rest_length_interval ="<<rest_length_interval ;
279
280 m_constraintIntervals.push_back(intervalDef);
281 }

284template<class DataTypes>
285void AdaptiveBeamLengthConstraint<DataTypes>::buildConstraintMatrix(const ConstraintParams* cParams, DataMatrixDeriv &c_d,
286 unsigned int &constraintId, const DataVecCoord & x_d)
287{
288 SOFA_UNUSED(cParams) ;
289 SOFA_UNUSED(x_d) ;
290
291 m_violations.clear();
292 Real constrainedLength = m_constrainedLength.getValue();
293
294 m_nbConstraints = 0;
295 m_cid = constraintId;
296
297 ReadAccessor<Data<VecCoord> > x = this->mstate->read(sofa::core::vec_id::read_access::position) ;
298 ReadAccessor<Data<VecCoord> > xfree = this->mstate->read(sofa::core::vec_id::read_access::freePosition) ;
299
300 auto c = sofa::helper::getWriteOnlyAccessor(c_d);
301
302 m_constraintIntervals.clear();
303 detectElongation( x.ref(), xfree.ref());
304
305 if( this->f_printLog.getValue())
306 {
307 stringstream tmp;
308 for (unsigned int i=0; i<m_constraintIntervals.size();i++)
309 {
310 tmp <<"constraint["<<i<<"] between pos: "<<m_constraintIntervals[i].posBegin<<" and pos: "<<m_constraintIntervals[i].posEnd << msgendl ;
311 }
312 msg_info() << tmp.str() ;
313 }
314
316 for (unsigned int i=0; i<m_constraintIntervals.size();i++)
317 {
319 unsigned int n0,n1;
320 n0=m_constraintIntervals[i].IdxBegin;
321 n1=m_constraintIntervals[i].IdxEnd;
322
324 Vec3 dir = m_constraintIntervals[i].posEnd - m_constraintIntervals[i].posBegin;
325 Vec3 PbPeFree= m_constraintIntervals[i].posFreeEnd - m_constraintIntervals[i].posFreeBegin;
326
327 dir.normalize();
328 Real length_free= dot(dir, PbPeFree);
329
331 m_violations.push_back(m_constraintIntervals[i].rest_length * constrainedLength - length_free);
332
334 Vec3 lever0 = x[n0].getOrientation().rotate( m_constraintIntervals[i].dof_H_begin.getOrigin() );
335 Vec3 lever1 = x[n1].getOrientation().rotate( m_constraintIntervals[i].dof_H_end.getOrigin() );
336
337 dmsg_info_when(lever0.norm() > 0.0001) << " lever0 ="<<lever0<<" dir ="<<dir ;
338 dmsg_info_when(lever1.norm() > 0.0001) << " lever1 ="<<lever1<<" -dir ="<<-dir ;
339
340 MatrixDerivRowIterator c_it = c.wref().writeLine(m_cid + m_nbConstraints);
341
342 c_it.addCol(n0, Vec6(dir, cross(lever0, dir) ) );
343 c_it.addCol(n1, Vec6(-dir, cross(lever1, -dir) ) );
344
345 m_nbConstraints++;
346
347 }
348 constraintId +=m_nbConstraints;
349 dmsg_info() << "end buildConstraintMatrix " ;
350}

375template<class DataTypes>
376void AdaptiveBeamLengthConstraint<DataTypes>::draw(const VisualParams* vparams)
377{
378 if (!vparams->displayFlags().getShowInteractionForceFields())
379 return;
380
381 if(m_constraintIntervals.size()==0)
382 return;
383
384 vparams->drawTool()->saveLastState();
385 vparams->drawTool()->setLightingEnabled(false);
386
387 std::vector< sofa::type::Vec3 > points;
388 std::vector< sofa::type::RGBAColor> colors;
389 points.reserve(m_constraintIntervals.size()*2);
390 colors.reserve(m_constraintIntervals.size());
391
392 for (unsigned int i = 0; i < m_constraintIntervals.size(); i++)
393 {
394 points.emplace_back(m_constraintIntervals[i].posBegin);
395 points.emplace_back(m_constraintIntervals[i].posEnd);
396
397 if (m_constraintIntervals[i].active)
398 colors.emplace_back(1.0f, 0.5f, 0.0f, 1.0f);
399 else
400 colors.emplace_back(0.0f, 1.0f, 0.0f, 1.0f);
401 }
402 vparams->drawTool()->drawPoints(points, 10, sofa::type::RGBAColor::green());
403 vparams->drawTool()->drawLines(points, 1.0f, colors);
404
405 vparams->drawTool()->restoreLastState();
406}
