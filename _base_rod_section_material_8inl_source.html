44template <class DataTypes>
45void BaseRodSectionMaterial<DataTypes>::init()
46{
47 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Loading);
48
49 // Prepare beam sections
50 double r = this->d_radius.getValue();
51 double rInner = this->d_innerRadius.getValue();
52 this->m_beamSection._r = r;
53 this->m_beamSection._rInner = rInner;
54 this->m_beamSection._Iz = M_PI * (r * r * r * r - rInner * rInner * rInner * rInner) / 4.0;
55 this->m_beamSection._Iy = this->m_beamSection._Iz;
56 this->m_beamSection._J = this->m_beamSection._Iz + this->m_beamSection._Iy;
57 this->m_beamSection._A = M_PI * (r * r - rInner * rInner);
58 this->m_beamSection._Asy = 0.0;
59 this->m_beamSection._Asz = 0.0;
60
61 // call delegate method to init the section
62 bool res = initSection();
63
64 if (res)
65 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Valid);
66 else
67 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Invalid);
68}
