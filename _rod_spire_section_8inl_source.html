55template <class DataTypes>
56void RodSpireSection<DataTypes>::getRestTransformOnX(Transform& global_H_local, const Real x_used, const Real x_start)
57{
58 Real projetedLength = d_spireDiameter.getValue() * M_PI;
59 Real lengthSpire = sqrt(d_spireHeight.getValue() * d_spireHeight.getValue() + projetedLength * projetedLength);
60 // angle in the z direction
61 Real phi = atan(d_spireHeight.getValue() / projetedLength);
62
63 Quat Qphi;
64 Qphi.axisToQuat(type::Vec3(0, 0, 1), phi);
65
66 // spire angle (if theta=2*PI, there is a complete spire between startx and x)
67 Real lengthCurve = x_used - x_start;
68 Real numSpire = lengthCurve / lengthSpire;
69 Real theta = 2 * M_PI * numSpire;
70
71 // computation of the Quat
72 Quat Qtheta;
73 Qtheta.axisToQuat(type::Vec3(0, 1, 0), theta);
74 Quat newSpireQuat = Qtheta * Qphi;
75
76 // computation of the position
77 Real radius = d_spireDiameter.getValue() / 2.0;
78 type::Vec3 PosEndCurve(radius * sin(theta), numSpire * d_spireHeight.getValue(), radius * (cos(theta) - 1));
79 type::Vec3 SpirePos = PosEndCurve + type::Vec3(x_start, 0, 0);
80
81 global_H_local.set(SpirePos, newSpireQuat);
82}
