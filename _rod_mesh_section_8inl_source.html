42template <class DataTypes>
43bool RodMeshSection<DataTypes>::initSection()
44{
45 // Get meshLoader, check first if loader has been set using link. Otherwise will search in current context.
46 p_loader = l_loader.get();
47
48 if (!p_loader)
49 this->getContext()->get(p_loader);
50
51 if (!p_loader) {
52 msg_error() << "Cannot find a mesh loader. Please insert a MeshObjLoader in the same node or use l_loader to specify the path in the scene graph.";
53 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Invalid);
54 return false;
55 }
56 else
57 {
58 msg_info() << "Found a mesh with " << p_loader->d_edges.getValue().size() << " edges";
59 return initFromLoader();
60 }
61}

64template <class DataTypes>
65void RodMeshSection<DataTypes>::getRestTransformOnX(Transform& global_H_local, const Real x_used, const Real x_start)
66{
67 Real abs_curr = x_used - x_start;
68 abs_curr = abs_curr /(this->d_length.getValue()) * m_absOfGeometry;
69
70 Coord p;
71
72 /*** find the range which includes the "requested" abs ***/
73 Real startingAbs = 0;
74 unsigned int index = 0;
75 while ((startingAbs < abs_curr) && (index < m_localRestPositions.size()))
76 {
77 index++;
78 startingAbs = m_curvAbs[index];
79 }
80
81 /*** OOB ***/
82 if (abs_curr > startingAbs)
83 {
84 msg_error() << "Out of bound position requested= " << abs_curr << " with startingAbs = " << startingAbs;
85 return;
86 }
87 else /*** Expected case ***/
88 {
89 const Real alpha = (abs_curr - m_curvAbs[index - 1]) / (m_curvAbs[index] - m_curvAbs[index - 1]);
90 const Real one_minus_alpha = 1 - alpha;
91 const type::Vec3 result = m_localRestTransforms[index - 1].getOrigin() * one_minus_alpha + m_localRestTransforms[index].getOrigin() * alpha;
92 Quat slerp;
93 slerp.slerp(m_localRestTransforms[index - 1].getOrientation(), m_localRestTransforms[index].getOrientation(), alpha, true);
94 slerp.normalize();
95
96 p.getCenter() = result;
97 p.getOrientation() = slerp;
98 }
99
100 type::Vec3 PosEndCurve = p.getCenter();
101 Quat ExtremityQuat = p.getOrientation();
102 type::Vec3 ExtremityPos = PosEndCurve + type::Vec3(x_start, 0, 0);
103
104 global_H_local.set(ExtremityPos, ExtremityQuat);
105}

108template <class DataTypes>
109bool RodMeshSection<DataTypes>::initFromLoader()
110{
111 if (!checkLoaderTopology())
112 {
113 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Invalid);
114 return false;
115 }
116
117 type::vector<type::Vec3> vertices;
118 sofa::core::topology::BaseMeshTopology::SeqEdges edges;
119
120 //get the topology position
121 auto topoVertices = sofa::helper::getReadAccessor(p_loader->d_positions);
122
123 //copy the topology edges in a local vector
124 auto topoEdges = sofa::helper::getReadAccessor(p_loader->d_edges);
125 edges = topoEdges.ref();
126
128 type::vector<unsigned int> verticesConnexion; //gives the number of edges connected to a vertex
129 for (unsigned int i = 0; i < topoVertices.size(); i++)
130 verticesConnexion.push_back(2);
131
132 for (const auto& ed : edges)
133 {
134 verticesConnexion[ed[0]]--;
135 verticesConnexion[ed[1]]--;
136 }
137
138 msg_info() << "Successfully compute the vertex connexion";
139
140 // check for the first corner of the edge
141 unsigned int firstIndex = 0;
142 bool found = false;
143 while ((firstIndex < verticesConnexion.size()) && !found)
144 {
145 if (verticesConnexion[firstIndex] == 1)
146 found = true;
147 else
148 firstIndex++;
149 }
150
151 if (firstIndex == verticesConnexion.size())
152 {
153 msg_error() << "The first vertex of the beam structure is not found, probably because of a closed structure";
154 this->d_componentState.setValue(sofa::core::objectmodel::ComponentState::Invalid);
155 return false;
156 }
157
158 vertices.push_back(topoVertices[firstIndex]);
159
160 while (edges.size() > 0)
161 {
162 auto it = edges.begin();
163 auto end = edges.end();
164
165 bool notFound = true;
166 while (notFound && (it != end))
167 {
168 const auto& ed = (*it);
169 auto toDel = it;
170 it++;
171 if (ed[0] == firstIndex)
172 {
173 vertices.push_back(topoVertices[ed[1]]);
174 firstIndex = ed[1];
175 edges.erase(toDel);
176 notFound = false;
177
178 }
179 else if (ed[1] == firstIndex)
180 {
181 vertices.push_back(topoVertices[ed[0]]);
182 firstIndex = ed[0];
183 edges.erase(toDel);
184 notFound = false;
185 }
186 }
187 }
188
189 msg_info() << "Successfully computed the topology";
190
191 m_localRestPositions = vertices;
192
193 initRestConfig();
194
195 return true;
196}

199template <class DataTypes>
200void RodMeshSection<DataTypes>::initRestConfig()
201{
202 m_curvAbs.clear();
203 Real tot = 0;
204 m_curvAbs.push_back(0);
205 Quat input, output;
206 input.identity();
207 m_localRestTransforms.resize(m_localRestPositions.size());
208 m_localRestTransforms[0].setOrigin(type::Vec3(0, 0, 0));
209 m_localRestTransforms[0].setOrientation(input);
210
211 for (unsigned int i = 0; i < m_localRestPositions.size() - 1; i++)
212 {
213 type::Vec3 vec = m_localRestPositions[i + 1] - m_localRestPositions[i];
214 Real norm = vec.norm();
215 tot += norm;
216
217 this->rotateFrameForAlignX(input, vec, output);
218
219 input = output;
220
221 m_localRestTransforms[i + 1].setOrientation(output);
222
223 type::Vec3 localPos = m_localRestPositions[i + 1] - m_localRestPositions[0];
224
225 m_localRestTransforms[i + 1].setOrigin(localPos);
226
227 m_curvAbs.push_back(tot);
228 }
229 m_absOfGeometry = tot;
230
231 this->d_length.setValue(m_absOfGeometry);
232
233 msg_info() << "Length of the loaded shape = " << m_absOfGeometry;
234}

237template <class DataTypes>
238bool RodMeshSection<DataTypes>::checkLoaderTopology()
239{
240 if (!p_loader->d_edges.getValue().size())
241 {
242 msg_error() << "There is no edges in the topology loaded by " << p_loader->getName();
243 return false;
244 }
245
246 if (p_loader->d_triangles.getValue().size())
247 {
248 msg_error() << "There are triangles in the topology loaded by " << p_loader->getName();
249 return false;
250 }
251
252 if (p_loader->d_quads.getValue().size())
253 {
254 msg_error() << "There are quads in the topology loaded by " << p_loader->getName();
255 return false;
256 }
257
258 if (p_loader->d_polygons.getValue().size())
259 {
260 msg_error() << "There are polygons in the topology loaded by " << p_loader->getName();
261 return false;
262 }
263
264 return true;
265}

268template <class DataTypes>
269void RodMeshSection<DataTypes>::rotateFrameForAlignX(const Quat& input, type::Vec3& x, Quat& output)
270{
271 x.normalize();
272 type::Vec3 x0 = input.inverseRotate(x);
273
274 Real cTheta = x0[0];
275 Real theta;
276 if (cTheta > (1 - EPSILON))
277 {
278 output = input;
279 }
280 else
281 {
282 theta = acos(cTheta);
283 // axis of rotation
284 type::Vec3 dw(0, -x0[2], x0[1]);
285 dw.normalize();
286
287 // computation of the rotation
288 Quat inputRoutput;
289 inputRoutput.axisToQuat(dw, theta);
290
291 output = input * inputRoutput;
292 }
293}
