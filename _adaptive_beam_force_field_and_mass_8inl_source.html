103template<class DataTypes>
104void AdaptiveBeamForceFieldAndMass<DataTypes>::computeStiffness(int beamId, BeamLocalMatrices& beamLocalMatrices)
105{
107 Real _G = beamLocalMatrices._youngM / (2.0 * (1.0 + beamLocalMatrices._cPoisson));
108
109 Real phiy, phiz;
110 Real L2 = (Real) (beamLocalMatrices._L * beamLocalMatrices._L);
111 Real L3 = (Real) (L2 * beamLocalMatrices._L);
112 Real EIy = (Real)(beamLocalMatrices._youngM * beamLocalMatrices._Iy);
113 Real EIz = (Real)(beamLocalMatrices._youngM * beamLocalMatrices._Iz);
114
116 if (beamLocalMatrices._Asy == 0)
117 phiy = 0.0;
118 else
119 phiy = (L2 == 0) ? 0.0 : (Real)(24.0 * (1.0 + beamLocalMatrices._cPoisson) * beamLocalMatrices._Iz / (beamLocalMatrices._Asy * L2));
120
121 if (beamLocalMatrices._Asz == 0)
122 phiz = 0.0;
123 else
124 phiz = (L2 == 0) ? 0.0 : (Real)(24.0 * (1.0 + beamLocalMatrices._cPoisson) * beamLocalMatrices._Iy / (beamLocalMatrices._Asz * L2));
125
126 beamLocalMatrices.m_K00.clear(); beamLocalMatrices.m_K01.clear(); beamLocalMatrices.m_K10.clear(); beamLocalMatrices.m_K11.clear();
127
129 beamLocalMatrices.m_K00[0][0] = beamLocalMatrices.m_K11[0][0] = (beamLocalMatrices._L == 0.0)? 0.0 : beamLocalMatrices._youngM * beamLocalMatrices._A/ beamLocalMatrices._L;
130 beamLocalMatrices.m_K00[1][1] = beamLocalMatrices.m_K11[1][1] = (L3 == 0.0)? 0.0 :(Real)(12.0*EIz/(L3*(1.0+phiy)));
131 beamLocalMatrices.m_K00[2][2] = beamLocalMatrices.m_K11[2][2] = (L3 == 0.0)? 0.0 : (Real)(12.0*EIy/(L3*(1.0+phiz)));
132 beamLocalMatrices.m_K00[3][3] = beamLocalMatrices.m_K11[3][3] = (beamLocalMatrices._L == 0.0)? 0.0 : _G* beamLocalMatrices._J/ beamLocalMatrices._L;
133 beamLocalMatrices.m_K00[4][4] = beamLocalMatrices.m_K11[4][4] = (beamLocalMatrices._L == 0.0)? 0.0 : (Real)((4.0+phiz)*EIy/(beamLocalMatrices._L*(1.0+phiz)));
134 beamLocalMatrices.m_K00[5][5] = beamLocalMatrices.m_K11[5][5] = (beamLocalMatrices._L == 0.0)? 0.0 : (Real)((4.0+phiy)*EIz/(beamLocalMatrices._L*(1.0+phiy)));
135
137 beamLocalMatrices.m_K00[4][2] =(L2 == 0.0)? 0.0 : (Real)(-6.0*EIy/(L2*(1.0+phiz)));
138 beamLocalMatrices.m_K00[5][1] =(L2 == 0.0)? 0.0 : (Real)( 6.0*EIz/(L2*(1.0+phiy)));
139 beamLocalMatrices.m_K11[5][1] = -beamLocalMatrices.m_K00[5][1];
140 beamLocalMatrices.m_K11[4][2] = -beamLocalMatrices.m_K00[4][2];
141
143 beamLocalMatrices.m_K10[0][0] = -beamLocalMatrices.m_K00[0][0];
144 beamLocalMatrices.m_K10[1][1] = -beamLocalMatrices.m_K00[1][1];
145 beamLocalMatrices.m_K10[1][5] = -beamLocalMatrices.m_K00[5][1];
146 beamLocalMatrices.m_K10[2][2] = -beamLocalMatrices.m_K00[2][2];
147 beamLocalMatrices.m_K10[2][4] = -beamLocalMatrices.m_K00[4][2];
148 beamLocalMatrices.m_K10[3][3] = -beamLocalMatrices.m_K00[3][3];
149 beamLocalMatrices.m_K10[4][2] = beamLocalMatrices.m_K00[4][2];
150 beamLocalMatrices.m_K10[4][4] =(beamLocalMatrices._L == 0.0)? 0.0 : (Real)((2.0-phiz)*EIy/(beamLocalMatrices._L*(1.0+phiz)));
151 beamLocalMatrices.m_K10[5][1] = beamLocalMatrices.m_K00[5][1];
152 beamLocalMatrices.m_K10[5][5] =(beamLocalMatrices._L == 0.0)? 0.0 : (Real)((2.0-phiy)*EIz/(beamLocalMatrices._L*(1.0+phiy)));
153
155 for (int i=0; i<=5; i++)
156 {
157 for (int j=i+1; j<6; j++)
158 {
159 beamLocalMatrices.m_K00[i][j] = beamLocalMatrices.m_K00[j][i];
160 beamLocalMatrices.m_K11[i][j] = beamLocalMatrices.m_K11[j][i];
161 }
162 }
163
165 beamLocalMatrices.m_K01 = beamLocalMatrices.m_K10.transposed();
166}

169template<class DataTypes>
170void AdaptiveBeamForceFieldAndMass<DataTypes>::computeMass(int beamId, BeamLocalMatrices& beamLocalMatrix)
171{
172 SOFA_UNUSED(beamId);
173 Real L2 = (Real) (beamLocalMatrix._L * beamLocalMatrix._L);
174 beamLocalMatrix.m_M00.clear(); beamLocalMatrix.m_M01.clear(); beamLocalMatrix.m_M10.clear(); beamLocalMatrix.m_M11.clear();
175
176 Real AL = beamLocalMatrix._A * beamLocalMatrix._L;
177 Real Iz_A = (beamLocalMatrix._A == 0.0) ? 0.0 : beamLocalMatrix._Iz / beamLocalMatrix._A;
178 Real Iy_A = (beamLocalMatrix._A == 0.0) ? 0.0 : beamLocalMatrix._Iy / beamLocalMatrix._A;
179
181 beamLocalMatrix.m_M00[0][0] = beamLocalMatrix.m_M11[0][0] = (Real)(1.0 / 3.0);
182 beamLocalMatrix.m_M00[1][1] = beamLocalMatrix.m_M11[1][1] = (L2 == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(13.0 / 35.0 + 6.0 * Iz_A / (5.0 * L2));
183 beamLocalMatrix.m_M00[2][2] = beamLocalMatrix.m_M11[2][2] = (L2 == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(13.0 / 35.0 + 6.0 * Iy_A / (5.0 * L2));
184 beamLocalMatrix.m_M00[3][3] = beamLocalMatrix.m_M11[3][3] = (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(beamLocalMatrix._J / (3.0 * beamLocalMatrix._A));
185 beamLocalMatrix.m_M00[4][4] = beamLocalMatrix.m_M11[4][4] = (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(L2 / 105.0 + 2 * Iy_A / 15.0);
186 beamLocalMatrix.m_M00[5][5] = beamLocalMatrix.m_M11[5][5] = (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(L2 / 105.0 + 2 * Iz_A / 15.0);
187
189 beamLocalMatrix.m_M00[4][2] = (beamLocalMatrix._L == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(-11.0 * beamLocalMatrix._L / 210.0 - beamLocalMatrix._Iy / (10 * AL));
190 beamLocalMatrix.m_M00[5][1] = (beamLocalMatrix._L == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(11.0 * beamLocalMatrix._L / 210.0 + beamLocalMatrix._Iz / (10 * AL));
191 beamLocalMatrix.m_M11[5][1] = -beamLocalMatrix.m_M00[5][1];
192 beamLocalMatrix.m_M11[4][2] = -beamLocalMatrix.m_M00[4][2];
193
194 beamLocalMatrix.m_M00 *= beamLocalMatrix._rho * AL;
195 beamLocalMatrix.m_M11 *= beamLocalMatrix._rho * AL;
196
198 beamLocalMatrix.m_M10[0][0] = (Real)(1.0 / 6.0);
199 beamLocalMatrix.m_M10[1][1] = (L2 == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(9.0 / 70.0 - 6.0 * Iz_A / (5.0 * L2));
200 beamLocalMatrix.m_M10[2][2] = (L2 == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(9.0 / 70.0 - 6.0 * Iy_A / (5.0 * L2));
201 beamLocalMatrix.m_M10[3][3] = (beamLocalMatrix._A == 0.0) ? 0.0: (Real)(beamLocalMatrix._J/(6.0*beamLocalMatrix._A));
202 beamLocalMatrix.m_M10[4][4] = (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(-L2 / 140.0 - Iy_A / 30.0);
203 beamLocalMatrix.m_M10[5][5] = (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(-L2 / 140.0 - Iz_A / 30.0);
204
205 beamLocalMatrix.m_M10[1][5] = (beamLocalMatrix._L == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(13 * beamLocalMatrix._L / 420.0 - beamLocalMatrix._Iz / (10.0 * AL));
206 beamLocalMatrix.m_M10[2][4] = (beamLocalMatrix._L == 0.0) || (beamLocalMatrix._A == 0.0) ? 0.0 : (Real)(-13 * beamLocalMatrix._L / 420.0 + beamLocalMatrix._Iy / (10.0 * AL));
207 beamLocalMatrix.m_M10[4][2] = -beamLocalMatrix.m_M10[2][4];
208 beamLocalMatrix.m_M10[5][1] = -beamLocalMatrix.m_M10[1][5];
209
210 beamLocalMatrix.m_M10 *= beamLocalMatrix._rho * AL;
211
213 for (int i=0; i<=5; i++)
214 {
215 for (int j=i+1; j<6; j++)
216 {
217 beamLocalMatrix.m_M00[i][j] = beamLocalMatrix.m_M00[j][i];
218 beamLocalMatrix.m_M11[i][j] = beamLocalMatrix.m_M11[j][i];
219 }
220 }
221
223 beamLocalMatrix.m_M01 = beamLocalMatrix.m_M10.transposed();
224}

227template<class DataTypes>
228void AdaptiveBeamForceFieldAndMass<DataTypes>::applyStiffnessLarge( VecDeriv& df, const VecDeriv& dx,
229 int bIndex, Index nd0Id, Index nd1Id,
230 SReal factor )
231{
232 if(nd0Id==nd1Id)
233 return;
234
235 Vec6NoInit U0, U1, u0, u1, f0, f1, F0, F1;
236 BeamLocalMatrices &beamLocalMatrix = m_localBeamMatrices[bIndex];
237
238 for (unsigned int i=0; i<6; i++)
239 {
240 U0[i] = dx[nd0Id][i];
241 U1[i] = dx[nd1Id][i];
242 }
243
245 u0 = beamLocalMatrix.m_A0Ref*U0;
246 u1 = beamLocalMatrix.m_A1Ref*U1;
247
249 f0 = beamLocalMatrix.m_K00*u0 + beamLocalMatrix.m_K01*u1;
250 f1 = beamLocalMatrix.m_K10*u0 + beamLocalMatrix.m_K11*u1;
251
253 F0 = beamLocalMatrix.m_A0Ref.multTranspose(f0);
254 F1 = beamLocalMatrix.m_A1Ref.multTranspose(f1);
255
257 for (unsigned int i=0; i<6; i++)
258 {
259 df[nd0Id][i] -= F0[i]*factor;
260 df[nd1Id][i] -= F1[i]*factor;
261 }
262}

265template<class DataTypes>
266void AdaptiveBeamForceFieldAndMass<DataTypes>::applyMassLarge( VecDeriv& df, int bIndex, Index nd0Id, Index nd1Id, SReal factor)
267{
268 const BeamLocalMatrices &beamLocalMatrix = m_localBeamMatrices[bIndex];
269
271 Vec6 a0 = beamLocalMatrix.m_A0Ref * m_gravity;
272 Vec6 a1 = beamLocalMatrix.m_A1Ref * m_gravity;
273
275 Vec6 f0 = beamLocalMatrix.m_M00*a0 + beamLocalMatrix.m_M01*a1;
276 Vec6 f1 = beamLocalMatrix.m_M10*a0 + beamLocalMatrix.m_M11*a1;
277
279 Vec6 F0 = beamLocalMatrix.m_A0Ref.multTranspose(f0);
280 Vec6 F1 = beamLocalMatrix.m_A1Ref.multTranspose(f1);
281
283 for (unsigned int i=0; i<6; i++)
284 {
285 df[nd0Id][i] += F0[i]*factor;
286 df[nd1Id][i] += F1[i]*factor;
287 }
288}

297void AdaptiveBeamForceFieldAndMass<DataTypes>::addMDx(const MechanicalParams* mparams , DataVecDeriv& dataf, const DataVecDeriv& datadx, SReal factor)
298{
299 SOFA_UNUSED(mparams);
300 SOFA_UNUSED(datadx);
301
302 auto f = sofa::helper::getWriteOnlyAccessor(dataf);
303
304 auto size = l_interpolation->getStateSize();
305 if (f.size() != size)
306 f.resize(size);
307
308 unsigned int numBeams = l_interpolation->getNumBeams();
309 for (unsigned int b=0; b<numBeams; b++)
310 {
311 unsigned int node0Idx, node1Idx;
312 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
313
314 applyMassLarge( f.wref(), b, node0Idx, node1Idx, factor);
315 }
316}

319template<class DataTypes>
320void AdaptiveBeamForceFieldAndMass<DataTypes>::addMToMatrix(const MechanicalParams *mparams,
321 const MultiMatrixAccessor* matrix)
322{
323 MultiMatrixAccessor::MatrixRef r = matrix->getMatrix(mstate);
324 Real mFact = (Real)mparams->mFactor();
325
326 unsigned int numBeams = l_interpolation->getNumBeams();
327
328 for (unsigned int b=0; b<numBeams; b++)
329 {
330 unsigned int node0Idx, node1Idx;
331 BeamLocalMatrices &bLM = m_localBeamMatrices[b];
332 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
333
335 Matrix6x6 M00 = bLM.m_A0Ref.multTranspose((bLM.m_M00 * bLM.m_A0Ref));
336 Matrix6x6 M01 = bLM.m_A0Ref.multTranspose((bLM.m_M01 * bLM.m_A1Ref));
337 Matrix6x6 M10 = bLM.m_A1Ref.multTranspose((bLM.m_M10 * bLM.m_A0Ref));
338 Matrix6x6 M11 = bLM.m_A1Ref.multTranspose((bLM.m_M11 * bLM.m_A1Ref));
339
340 int index0[6], index1[6];
341 for (int i=0;i<6;i++)
342 index0[i] = r.offset+node0Idx*6+i;
343 for (int i=0;i<6;i++)
344 index1[i] = r.offset+node1Idx*6+i;
345
346 for (int i=0;i<6;i++)
347 {
348 for (int j=0;j<6;j++)
349 {
350 r.matrix->add(index0[i], index0[j], M00(i,j)*mFact);
351 r.matrix->add(index0[i], index1[j], M01(i,j)*mFact);
352 r.matrix->add(index1[i], index0[j], M10(i,j)*mFact);
353 r.matrix->add(index1[i], index1[j], M11(i,j)*mFact);
354 }
355 }
356
357 }

360template<class DataTypes>
361void AdaptiveBeamForceFieldAndMass<DataTypes>::buildMassMatrix(sofa::core::behavior::MassMatrixAccumulator* matrices)
362{
363 const unsigned int numBeams = l_interpolation->getNumBeams();
364
365
366 for (unsigned int b=0; b<numBeams; b++)
367 {
368 unsigned int node0Idx, node1Idx;
369 BeamLocalMatrices &bLM = m_localBeamMatrices[b];
370 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
371
373 const Matrix6x6 M00 = bLM.m_A0Ref.multTranspose(bLM.m_M00 * bLM.m_A0Ref);
374 const Matrix6x6 M01 = bLM.m_A0Ref.multTranspose(bLM.m_M01 * bLM.m_A1Ref);
375 const Matrix6x6 M10 = bLM.m_A1Ref.multTranspose(bLM.m_M10 * bLM.m_A0Ref);
376 const Matrix6x6 M11 = bLM.m_A1Ref.multTranspose(bLM.m_M11 * bLM.m_A1Ref);
377
378
379 matrices->add(node0Idx * 6, node0Idx * 6, M00);
380 matrices->add(node0Idx * 6, node1Idx * 6, M01);
381 matrices->add(node1Idx * 6, node0Idx * 6, M10);
382 matrices->add(node1Idx * 6, node1Idx * 6, M11);
383 }

387template<class DataTypes>
388void AdaptiveBeamForceFieldAndMass<DataTypes>::addMBKToMatrix(const MechanicalParams* mparams,
389 const MultiMatrixAccessor* matrix)
390{
391 MultiMatrixAccessor::MatrixRef r = matrix->getMatrix(mstate);
392 Real kFact = (Real)mparams->kFactor();
393 Real mFact = (Real)mparams->mFactor();
394
395 Real totalMass = 0;
396 unsigned int numBeams = l_interpolation->getNumBeams();
397
398 for (unsigned int b=0; b<numBeams; b++)
399 {
400 unsigned int node0Idx, node1Idx;
401 const BeamLocalMatrices &bLM = m_localBeamMatrices[b];
402 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
403
404 int index0[6], index1[6];
405 for (int i=0;i<6;i++)
406 index0[i] = r.offset+node0Idx*6+i;
407 for (int i=0;i<6;i++)
408 index1[i] = r.offset+node1Idx*6+i;
409
410 if(node0Idx!=node1Idx) // no rigidification
411 {
412 // matrices in global frame
413 Matrix6x6 K00 = bLM.m_A0Ref.multTranspose((bLM.m_K00 * bLM.m_A0Ref));
414 Matrix6x6 K01 = bLM.m_A0Ref.multTranspose((bLM.m_K01 * bLM.m_A1Ref));
415 Matrix6x6 K10 = bLM.m_A1Ref.multTranspose((bLM.m_K10 * bLM.m_A0Ref));
416 Matrix6x6 K11 = bLM.m_A1Ref.multTranspose((bLM.m_K11 * bLM.m_A1Ref));
417
418 for (int i=0;i<6;i++)
419 {
420 for (int j=0;j<6;j++)
421 {
422 r.matrix->add(index0[i], index0[j], - K00(i,j)*kFact);
423 r.matrix->add(index0[i], index1[j], - K01(i,j)*kFact);
424 r.matrix->add(index1[i], index0[j], - K10(i,j)*kFact);
425 r.matrix->add(index1[i], index1[j], - K11(i,j)*kFact);
426 }
427 }
428 }
429
430 // matrices in global frame
431 Matrix6x6 M00 = bLM.m_A0Ref.multTranspose((bLM.m_M00 * bLM.m_A0Ref));
432 Matrix6x6 M01 = bLM.m_A0Ref.multTranspose((bLM.m_M01 * bLM.m_A1Ref));
433 Matrix6x6 M10 = bLM.m_A1Ref.multTranspose((bLM.m_M10 * bLM.m_A0Ref));
434 Matrix6x6 M11 = bLM.m_A1Ref.multTranspose((bLM.m_M11 * bLM.m_A1Ref));
435
436 for (int i=0;i<6;i++)
437 {
438 for (int j=0;j<6;j++)
439 {
440 totalMass += M00(i,j)*mFact + M11(i,j)*mFact;
441 r.matrix->add(index0[i], index0[j], M00(i,j)*mFact);
442 r.matrix->add(index0[i], index1[j], M01(i,j)*mFact);
443 r.matrix->add(index1[i], index0[j], M10(i,j)*mFact);
444 r.matrix->add(index1[i], index1[j], M11(i,j)*mFact);
445 }
446 }
447 }

461template<class DataTypes>
462void AdaptiveBeamForceFieldAndMass<DataTypes>::addForce (const MechanicalParams* mparams ,
463 DataVecDeriv& dataf,
464 const DataVecCoord& datax,
465 const DataVecDeriv& v)
466{
467 SCOPED_TIMER("AdaptiveBeamForceFieldAndMass_addForce");
468 SOFA_UNUSED(v);
469
470 auto f = sofa::helper::getWriteOnlyAccessor(dataf);
471 const VecCoord& x = datax.getValue();
472
473 f.resize(x.size()); // current content of the vector will remain the same (http://www.cplusplus.com/reference/vector/vector/resize/)
474
475 unsigned int numBeams = l_interpolation->getNumBeams();
476 m_localBeamMatrices.resize(numBeams);
477
478 if(d_computeMass.getValue())
479 computeGravityVector();
480
487 for (unsigned int beamId=0; beamId <numBeams; beamId++)
488 {
490 sofa::Index node0Idx, node1Idx;
491 l_interpolation->getNodeIndices(beamId, node0Idx, node1Idx);
492
494 BeamLocalMatrices& beamMatrices = m_localBeamMatrices[beamId];
495
497 Transform global_H_local0, global_H_local1;
498
500 l_interpolation->computeTransform(beamId, node0Idx, node1Idx, global_H_local0, global_H_local1, x);
501
503 Transform global_H_local;
504 Real baryX = 0.5;
505 Real L = l_interpolation->getLength(beamId);
506
507 l_interpolation->InterpolateTransformUsingSpline(global_H_local, baryX, global_H_local0, global_H_local1, L);
508
509 Transform local_H_local0 = global_H_local.inversed()*global_H_local0;
510 Transform local_H_local1 = global_H_local.inversed()*global_H_local1;
511
513 Transform DOF0_H_local0, DOF1_H_local1;
514 l_interpolation->getDOFtoLocalTransform(beamId, DOF0_H_local0, DOF1_H_local1);
515
517 Transform global_R_DOF0(Vec3(0,0,0), x[node0Idx].getOrientation());
518 Transform global_R_DOF1(Vec3(0,0,0), x[node1Idx].getOrientation());
520 global_H_local0 = global_R_DOF0*DOF0_H_local0;
521 global_H_local1 = global_R_DOF1*DOF1_H_local1;
522
523 Transform DOF0global_H_Node0local, DOF1global_H_Node1local;
524
525 DOF0global_H_Node0local.set(global_H_local0.getOrigin(), global_H_local.getOrientation() );
526 DOF1global_H_Node1local.set(global_H_local1.getOrigin(), global_H_local.getOrientation() );
527
528 //TODO(dmarchal 2017-05-17) Please specify who/when this will be done
529 //TODO A verifier : global_H_local0.getOrigin() == x[node0Idx].getOrientation().rotate(DOF0_H_local0.getOrigin())
530
532 beamMatrices.m_A0Ref = DOF0global_H_Node0local.inversed().getAdjointMatrix();
533 beamMatrices.m_A1Ref = DOF1global_H_Node1local.inversed().getAdjointMatrix();
534
536
538
540 auto beamInterpolation = l_interpolation.get();
541 beamInterpolation->getInterpolationParameters(beamId, beamMatrices._L, beamMatrices._A, beamMatrices._Iy,
542 beamMatrices._Iz, beamMatrices._Asy, beamMatrices._Asz, beamMatrices._J);
543
544 beamMatrices._rho = d_massDensity.getValue(); // for BeamInterpolation which is not overidding the _rho
545 beamInterpolation->getMechanicalParameters(beamId, beamMatrices._youngM, beamMatrices._cPoisson, beamMatrices._rho);
546
548 if(d_computeMass.getValue())
549 {
550 computeMass(beamId, beamMatrices);
551 }
552
554 if(node0Idx==node1Idx)
555 continue;
556
558 computeStiffness(beamId, beamMatrices);
559
563 Transform local_H_local0_rest,local_H_local1_rest;
564 l_interpolation->getSplineRestTransform(beamId, local_H_local0_rest, local_H_local1_rest);
565
567 SpatialVector u0 = local_H_local0.CreateSpatialVector() - local_H_local0_rest.CreateSpatialVector();
568 SpatialVector u1 = local_H_local1.CreateSpatialVector() - local_H_local1_rest.CreateSpatialVector();
569
571 Vec6NoInit U0local, U1local;
572
573 for (unsigned int i=0; i<3; i++)
574 {
575 U0local[i] = u0.getLinearVelocity()[i];
576 U0local[i+3] = u0.getAngularVelocity()[i];
577 U1local[i] = u1.getLinearVelocity()[i];
578 U1local[i+3] = u1.getAngularVelocity()[i];
579 }
580
581 if(d_reinforceLength.getValue())
582 {
583 Vec3 P0,P1,P2,P3;
584 Real length;
585 Real rest_length = l_interpolation->getLength(beamId);
586 l_interpolation->getSplinePoints(beamId, x, P0, P1, P2, P3);
587 l_interpolation->computeActualLength(length, P0, P1, P2, P3);
588
589 U0local[0]=(-length+rest_length)/2;
590 U1local[0]=( length-rest_length)/2;
591 }
592
593 if (!d_useShearStressComputation.getValue())
594 {
597 Vec3 ResultNode0, ResultNode1;
598 l_interpolation->computeStrechAndTwist(beamId, x, ResultNode0, ResultNode1);
599
600 Real ux0 =-ResultNode0[0] + l_interpolation->getLength(beamId)/2;
601 Real ux1 = ResultNode1[0] - l_interpolation->getLength(beamId)/2;
602
603 U0local[0] = ux0;
604 U1local[0] = ux1;
605
606 U0local[3] =-ResultNode0[2];
607 U1local[3] = ResultNode1[2];
608
610 }
611
613 Vec6 f0 = beamMatrices.m_K00 * U0local + beamMatrices.m_K01 * U1local;
614 Vec6 f1 = beamMatrices.m_K10 * U0local + beamMatrices.m_K11 * U1local;
615
617 Vec6 F0_ref = beamMatrices.m_A0Ref.multTranspose(f0);
618 Vec6 F1_ref = beamMatrices.m_A1Ref.multTranspose(f1);
619
621 for (unsigned int i=0; i<6; i++)
622 {
623 f[node0Idx][i]-=F0_ref[i];
624 f[node1Idx][i]-=F1_ref[i];
625 }
626 }
627
628 if(d_computeMass.getValue())
629 {
631 DataVecDeriv emptyVec;
632 addMDx(mparams, dataf, emptyVec, 1.0);
633 }
634}

660template<class DataTypes>
661void AdaptiveBeamForceFieldAndMass<DataTypes>::addKToMatrix(const MechanicalParams* mparams,
662 const MultiMatrixAccessor* matrix)
663{
664 MultiMatrixAccessor::MatrixRef matrixRef = matrix->getMatrix(mstate);
665 Real k = (Real)mparams->kFactor();
666
667 unsigned int numBeams = l_interpolation->getNumBeams();
668
669 for (unsigned int b=0; b<numBeams; b++)
670 {
671 unsigned int node0Idx, node1Idx;
672 const BeamLocalMatrices &beamLocalMatrix = m_localBeamMatrices[b];
673 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
674
675 if(node0Idx==node1Idx)
676 continue;
677
678 // matrices in global frame
679 Matrix6x6 K00 = beamLocalMatrix.m_A0Ref.multTranspose((beamLocalMatrix.m_K00 * beamLocalMatrix.m_A0Ref));
680 Matrix6x6 K01 = beamLocalMatrix.m_A0Ref.multTranspose((beamLocalMatrix.m_K01 * beamLocalMatrix.m_A1Ref));
681 Matrix6x6 K10 = beamLocalMatrix.m_A1Ref.multTranspose((beamLocalMatrix.m_K10 * beamLocalMatrix.m_A0Ref));
682 Matrix6x6 K11 = beamLocalMatrix.m_A1Ref.multTranspose((beamLocalMatrix.m_K11 * beamLocalMatrix.m_A1Ref));
683
684 int index0[6], index1[6];
685 for (int i=0;i<6;i++)
686 index0[i] = matrixRef.offset+node0Idx*6+i;
687 for (int i=0;i<6;i++)
688 index1[i] = matrixRef.offset+node1Idx*6+i;
689
690 for (int i=0;i<6;i++)
691 {
692 for (int j=0;j<6;j++)
693 {
694 matrixRef.matrix->add(index0[i], index0[j], - K00(i,j)*k);
695 matrixRef.matrix->add(index0[i], index1[j], - K01(i,j)*k);
696 matrixRef.matrix->add(index1[i], index0[j], - K10(i,j)*k);
697 matrixRef.matrix->add(index1[i], index1[j], - K11(i,j)*k);
698 }
699 }
700 }

704void AdaptiveBeamForceFieldAndMass<DataTypes>::buildStiffnessMatrix(core::behavior::StiffnessMatrix* matrix)
705{
706 const unsigned int numBeams = l_interpolation->getNumBeams();
707
708
709 auto dfdx = matrix->getForceDerivativeIn(this->mstate)
710 .withRespectToPositionsIn(this->mstate);
711
712 for (unsigned int b=0; b<numBeams; b++)
713 {
714 unsigned int node0Idx, node1Idx;
715 const BeamLocalMatrices &beamLocalMatrix = m_localBeamMatrices[b];
716 l_interpolation->getNodeIndices( b, node0Idx, node1Idx );
717
718 if(node0Idx==node1Idx)
719 continue;
720
721 // matrices in global frame
722 Matrix6x6 K00 = beamLocalMatrix.m_A0Ref.multTranspose(beamLocalMatrix.m_K00 * beamLocalMatrix.m_A0Ref);
723 Matrix6x6 K01 = beamLocalMatrix.m_A0Ref.multTranspose(beamLocalMatrix.m_K01 * beamLocalMatrix.m_A1Ref);
724 Matrix6x6 K10 = beamLocalMatrix.m_A1Ref.multTranspose(beamLocalMatrix.m_K10 * beamLocalMatrix.m_A0Ref);
725 Matrix6x6 K11 = beamLocalMatrix.m_A1Ref.multTranspose(beamLocalMatrix.m_K11 * beamLocalMatrix.m_A1Ref);
726
727
728 dfdx(node0Idx*6, node0Idx*6) += - K00;
729 dfdx(node0Idx*6, node1Idx*6) += - K01;
730 dfdx(node1Idx*6, node0Idx*6) += - K10;
731 dfdx(node1Idx*6, node1Idx*6) += - K11;
732 }
